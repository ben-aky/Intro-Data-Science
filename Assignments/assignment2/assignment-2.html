# clean v-dem data
vdem_dta <- vdem_dta |>    
  select(!country_text_id) |>
  rename (country = country_name)  |>
  mutate(iso3c = countrycode(sourcevar = country_id,
        origin = "vdem",

  relocate(iso3c, .after = country_id)  |>
  select(!country_id)
# clean wb data
wb_dta <- wb_dta |>
